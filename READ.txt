4. Best practices
   - Use camelCase for component names
      example:
      - Navbar
      - Main
      - Footer
      - About
      - Contact
      - Error
   - Use PascalCase for class names
      example:
      - .Navbar
      - .Main
      - .Footer
      - .About
      - .Contact
      - .Error
   - Use kebab-case for CSS class names
      example:
      - .navbar
      - .main
      - .footer
      - .about
      - .contact
      - .error
   - Use snake_case for JavaScript variables
      example:
      - const navbar = <Navbar />;
      - const main = <Main />;
      - const footer = <Footer />;
      - const about = <About />;
      - const contact = <Contact />;
      - const error = <Error />;
   - Use snake_case for JavaScript constants
      example:
      - const NAVBAR = <Navbar />;
      - const MAIN = <Main />;
      - const FOOTER = <Footer />;
      - const ABOUT = <About />;
      - const CONTACT = <Contact />;
      - const ERROR = <Error />;
   - Use camelCase for JavaScript functions
      example:
      - function navbar() {
      - function main() {
      - function footer() {
      - function about() {
      - function contact() {
      - function error() {
   - Use camelCase for JavaScript event handlers
      example:
      - function handleClick() {
      - function handleChange() {
      - function handleFocus() {
      - function handleBlur() {
      - function handleKeyPress() {
   - Use camelCase for JavaScript state variables
      example:
      - const [name, setName] = useState('');
      - const [email, setEmail] = useState('');
   - Use PascalCase for JavaScript classes
      example:
      - class Navbar extends React.Component {
      - class Main extends React.Component {
      - class Footer extends React.Component {
      - class About extends React.Component {
      - class Contact extends React.Component {
      - class Error extends React.Component {
   - Use camelCase for JavaScript modules
      example:
      - import { useState } from 'react';
      - import { BrowserRouter as Router, Routes, Route } from 'react-router-dom';
      - import Navbar from './components/Navbar';
      - import Main from './components/Main';
      - import Footer from './components/Footer';
